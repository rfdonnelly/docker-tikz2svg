= docker-tikz2svg

A Docker image that converts TikZ picture to an SVG.

== Build

 docker build -t tikz2svg .

== Run

[source,sh]
----
docker run \
    --rm \
    --user $(id -u):$(id -g) \
    --volume "$PWD":/src \
    --workdir /src \
    tikz2svg \
    latex example.tex

docker run \
    --rm \
    --user $(id -u):$(id -g) \
    --volume "$PWD":/src \
    --workdir /src \
    tikz2svg \
    dvisvgm --font-format=woff example.dvi

file example.svg
----
